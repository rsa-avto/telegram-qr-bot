stages:
  - build
  - deploy

variables:
  IMAGE_NAME: rsa-avto-bot
  DOCKER_TAG: latest

# Сборка Docker-образа
build:
  stage: build
  image: docker:24.0.5
  services:
    - docker:dind
  script:
    - echo "Building Docker image..."
    - docker build -t $IMAGE_NAME:$DOCKER_TAG .
  only:
    - master

# Деплой контейнера
deploy:
  stage: deploy
  image: docker:24.0.5
  services:
    - docker:dind
  script:
    - echo "Stopping existing container if exists..."
    - docker stop $IMAGE_NAME || true
    - docker rm $IMAGE_NAME || true
    - echo "Running new container..."
    - docker run -d --name $IMAGE_NAME -p 8000:8000 $IMAGE_NAME:$DOCKER_TAG
  only:
    - master
